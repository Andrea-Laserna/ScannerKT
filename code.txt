# A simple program demonstrating variables and scope
var greeting = "Hello";
var name = "World";

print greeting + ", " + name + "!";

var x = 10;
var y = 20;

{
    var x = 5;  # shadows outer x
    print "Inner x: " + x;
    print "Outer y: " + y;
}

print "Outer x: " + x;

# Reassignment
x = x + y;
print "x + y = " + x;
