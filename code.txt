# A simple program demonstrating variables, concatenation, and scope

RES_STR greeting
RES_STR name
RES_STR msg

MOVE greeting, "Hello"
MOVE name, "World"

# Print a single value
PRINT greeting

# Concatenation example
PRINT CONCAT greeting, CONCAT ", ", CONCAT name, CONCAT "!"

RES_INT x
RES_INT y

MOVE x, 10
MOVE y, 20

# Separate block with a new local x (shadowing) written in Bridge style
{
    RES_INT x
    MOVE x, 5
    MOVE msg, CONCAT "Inner x: ", x
    PRINT msg
    MOVE msg, CONCAT "Outer y: ", y
    PRINT msg
}

MOVE msg, CONCAT "Outer x: ", x
PRINT msg

# Reassignment using ADD_OP (or keep + if supported by your grammar)
MOVE x, ADD x, y
MOVE msg, CONCAT "x + y = ", x
PRINT msg
